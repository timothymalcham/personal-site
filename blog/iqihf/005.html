class bst {

  constructor() {
    this.root = null;
  }

  add(value) {

    // create a new node that we'll soon add to the tree
    let node = {
      value,
      left: null,
      right: null
    }

    let current;

    if (this.head === null) {
      this.root = node
    } else {

      current = this.root

      while (true) {

        if (value < current.value) {

          if (current.left === null) {
            current.left = node;
            break;
          } else {
            current = current.left;
          }
        } else if (value > current.value) {

          if (current.right === null) {
            current.right = node;
            break;
          } else {
            current = current.right;
          }
        } else {
          break;
        }
      }
    }
  }
}
