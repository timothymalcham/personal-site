Interview Questions I've Failed, 002: JS Debounce Function

function debounce(func, wait) {
  let timeout;
  // our debounce function returns a function
  return () => {
    let context = this;
    let args = arguments;
    let later = () => {
      timeout = null;
      // run the function passed as an argument to our debounce
      func.apply(context, args);
    }
    clearTimeout(timeout)
    // set a timer, pass in the applied function,
    // and the designated wait time.
    timeout = setTimeout(later, wait);
  }
}
const myFunction = debounce(() => {
  console.log('meow');
}, 250);
window.addEventListener('resize', myFunction);
